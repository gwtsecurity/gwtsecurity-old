004    package com.gwt.ss.requestfactory.client.loginable;
005    
006    import com.google.gwt.core.client.GWT;
007    import com.google.gwt.http.client.Request;
008    import com.google.gwt.http.client.RequestBuilder;
009    import com.google.gwt.http.client.RequestCallback;
010    import com.google.gwt.http.client.RequestException;
011    import com.google.gwt.http.client.Response;
012    import com.google.gwt.user.client.rpc.SerializationStreamFactory;
013    import com.google.web.bindery.requestfactory.gwt.client.DefaultRequestTransport;
014    import com.google.web.bindery.requestfactory.shared.ServerFailure;
015    import com.gwt.ss.client.GwtLogin;
016    import com.gwt.ss.client.exceptions.GwtAccessDeniedException;
017    import com.gwt.ss.client.exceptions.GwtSecurityException;
018    import com.gwt.ss.client.loginable.AbstractLoginHandler;
019    import com.gwt.ss.client.loginable.HasLoginHandler;
020    import com.gwt.ss.client.loginable.LoginCancelException;
021    import com.gwt.ss.client.loginable.LoginHandler;
022    import com.gwt.ss.shared.GwtConst;
023    
024    /**
025     * Intercepts the requestfactory requests and starts the login process if necessary.
026     * 
027     * @version $Rev: 237 $
028     * @author Steven Jardine
029     */
030    public class LoginableRequestTransport extends DefaultRequestTransport {
031    
032        private class LoginableRequestCallback implements RequestCallback {
033    
034            private final String payload;
035    
036            private final TransportReceiver receiver;
037    
038            public LoginableRequestCallback(final String payload, final TransportReceiver receiver) {
039                this.payload = payload;
040                this.receiver = receiver;
041            }
042    
043            /** {@inheritDoc} */
044            @Override
045            public void onError(final Request request, final Throwable exception) {
046                receiver.onTransportFailure(new ServerFailure(exception.getMessage(), exception.getClass().getName(), null,
047                    false));
048            }
049    
050            /** {@inheritDoc} */
051            @Override
052            public void onResponseReceived(final Request request, final Response response) {
053                if (Response.SC_OK == response.getStatusCode()) {
054                    String responsePayload = response.getText();
055                    GwtSecurityException caught = deserializeSecurityException(responsePayload);
056                    if (loginHandler != null && caught != null) {
057                        if (caught.isAuthenticated() && caught instanceof GwtAccessDeniedException) {
058                            onError(request, caught);
059                        } else {
060                            LoginHandler lh = new AbstractLoginHandler() {
061                                @Override
062                                public void onCancelled() {
063                                    onError(request, new LoginCancelException("Login Cancelled"));
064                                }
065    
066                                @Override
067                                public void resendPayload() {
068                                    send(payload, receiver);
069                                }
070                            };
071                            lh.setLoginHandlerRegistration(loginHandler.addLoginHandler(lh));
072                            loginHandler.startLogin(caught);
073                        }
074                    } else {
075                        receiver.onTransportSuccess(responsePayload);
076                    }
077                } else {
078                    String message = response.getStatusCode() + " " + response.getText();
079                    receiver.onTransportFailure(new ServerFailure(message));
080                }
081            }
082        }
083    
084        private static final String EXCEPTION_PREFIX = "//EX[";
085    
086        private static SerializationStreamFactory streamFactory = null;
087    
088        /**
089         * Deserialize the rpc exception.
090         * 
091         * @param payload the exception payload to deserialize.
092         * @return the deserialized security exception.
093         */
094        private static GwtSecurityException deserializeSecurityException(String payload) {
095            if (payload != null && payload.indexOf(EXCEPTION_PREFIX) != -1) {
096                try {
097                    if (streamFactory == null) {
098                        streamFactory = GWT.create(GwtLogin.class);
099                    }
100                    return (GwtSecurityException) streamFactory.createStreamReader(payload).readObject();
101                } catch (Exception e) {
102                    GWT.log(e.getMessage(), e);
103                }
104            }
105            return null;
106        }
107    
108        private final HasLoginHandler loginHandler;
109    
110        /**
111         * Constructor.
112         * 
113         * @param loginHandler the login handler.
114         */
115        public LoginableRequestTransport(final HasLoginHandler loginHandler) {
116            super();
117            this.loginHandler = loginHandler;
118        }
119    
120        /** {@inheritDoc} */
121        @Override
122        public void send(final String payload, final TransportReceiver receiver) {
123            RequestBuilder builder = createRequestBuilder();
124            builder.setHeader(GwtConst.GWT_RF_HEADER, "true");
125            configureRequestBuilder(builder);
126            builder.setRequestData(payload);
127            builder.setCallback(new LoginableRequestCallback(payload, receiver));
128            try {
129                builder.send();
130            } catch (RequestException e) {
131                GWT.log(e.getMessage(), e);
132            }
133        }
134    
135    }
