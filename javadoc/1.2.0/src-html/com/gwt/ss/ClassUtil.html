004    package com.gwt.ss;
005    
006    import java.io.File;
007    import java.io.IOException;
008    import java.net.URL;
009    import java.util.ArrayList;
010    import java.util.Enumeration;
011    import java.util.List;
012    import java.util.Set;
013    import java.util.TreeSet;
014    import java.util.zip.ZipEntry;
015    import java.util.zip.ZipInputStream;
016    
017    import org.slf4j.Logger;
018    import org.slf4j.LoggerFactory;
019    
020    /**
021     * Utility class to find classes in a package.
022     * 
023     * @version $Rev: 220 $
024     * @author Steven Jardine
025     */
026    public final class ClassUtil {
027    
028        private static final Logger LOG = LoggerFactory.getLogger(ClassUtil.class);
029    
030        /**
031         * Recursive method used to find all classes in a given directory and subdirs. Adapted from
032         * http://snippets.dzone.com/posts/show/4831 and extended to support use of JAR files
033         * 
034         * @param directory The base directory
035         * @param packageName The package name for classes found inside the base directory
036         * @return The classes
037         * @throws ClassNotFoundException
038         */
039        private static Set<String> findClasses(String directory, String packageName) throws Exception {
040            Set<String> classes = new TreeSet<String>();
041            if (directory.startsWith("file:") && directory.contains("!")) {
042                String[] split = directory.split("!");
043                URL jar = new URL(split[0]);
044                ZipInputStream zip = new ZipInputStream(jar.openStream());
045                try {
046                    ZipEntry entry = null;
047                    while ((entry = zip.getNextEntry()) != null) {
048                        if (entry.getName().endsWith(".class")) {
049                            String className = entry.getName().replaceAll("[$].*", "").replaceAll("[.]class", "")
050                                .replace('/', '.');
051                            if (className.startsWith(packageName)) {
052                                classes.add(className);
053                            }
054                        }
055                    }
056                } finally {
057                    zip.close();
058                }
059            }
060            File dir = new File(directory);
061            if (!dir.exists()) { return classes; }
062            File[] files = dir.listFiles();
063            for (File file : files) {
064                if (file.isDirectory()) {
065                    assert !file.getName().contains(".");
066                    classes.addAll(findClasses(file.getAbsolutePath(), packageName + "." + file.getName()));
067                } else if (file.getName().endsWith(".class")) {
068                    classes.add(packageName + '.' + file.getName().substring(0, file.getName().length() - 6));
069                }
070            }
071            return classes;
072        }
073    
074        /**
075         * Scans all classes accessible from the context class loader which belong to the given package and
076         * subpackages. Adapted from http://snippets.dzone.com/posts/show/4831 and extended to support use of JAR
077         * files
078         * 
079         * @param packageName The base package
080         * @return The classes
081         * @throws ClassNotFoundException
082         * @throws IOException
083         */
084        public static Class<?>[] getClasses(String packageName) {
085            try {
086                ClassLoader classLoader = Thread.currentThread().getContextClassLoader();
087                assert classLoader != null;
088                String path = packageName.replace('.', '/');
089                Enumeration<URL> resources = classLoader.getResources(path);
090                List<String> dirs = new ArrayList<String>();
091                while (resources.hasMoreElements()) {
092                    URL resource = resources.nextElement();
093                    dirs.add(resource.getFile());
094                }
095                TreeSet<String> classes = new TreeSet<String>();
096                for (String directory : dirs) {
097                    classes.addAll(findClasses(directory, packageName));
098                }
099                ArrayList<Class<?>> classList = new ArrayList<Class<?>>();
100                for (String clazz : classes) {
101                    classList.add(Class.forName(clazz));
102                }
103                return classList.toArray(new Class[classes.size()]);
104            } catch (Exception e) {
105                LOG.error(e.getMessage(), e);
106                return null;
107            }
108        }
109    
110        /**
111         * Private constructor.
112         */
113        private ClassUtil() {
114            super();
115        }
116    
117    }
