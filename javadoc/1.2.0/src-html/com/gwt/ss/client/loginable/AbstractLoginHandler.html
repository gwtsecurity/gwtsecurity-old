005    package com.gwt.ss.client.loginable;
006    
007    import com.google.gwt.core.client.GWT;
008    import com.google.gwt.core.client.Scheduler;
009    import com.google.web.bindery.event.shared.HandlerRegistration;
010    
011    /**
012     * Common login request code for
013     * 
014     * @version $Rev: 174 $
015     * @author Steven Jardine
016     */
017    public abstract class AbstractLoginHandler implements LoginHandler {
018    
019        /**
020         * Cancelled message text.
021         */
022        public static final String CANCELLED_MSG = "User cancelled login process.";
023    
024        private HandlerRegistration registration;
025    
026        /**
027         * User cancelled login process.
028         */
029        public abstract void onCancelled();
030    
031        /** {@inheritDoc} */
032        @Override
033        public void onLogin(final LoginEvent e) {
034            if (registration != null) {
035                registration.removeHandler();
036            }
037            GWT.log("Receive login " + (e.isCanceled() ? "cancel" : "succeed") + " event!");
038            if (e.isCanceled()) {
039                onCancelled();
040            } else {
041                Scheduler.get().scheduleDeferred(new Scheduler.ScheduledCommand() {
042                    @Override
043                    public void execute() {
044                        resendPayload();
045                    }
046                });
047            }
048        }
049    
050        /**
051         * Resend the payload for this request.
052         */
053        public abstract void resendPayload();
054    
055        /** {@inheritDoc} */
056        @Override
057        public void setLoginHandlerRegistration(final HandlerRegistration registration) {
058            this.registration = registration;
059        }
060    
061    }
