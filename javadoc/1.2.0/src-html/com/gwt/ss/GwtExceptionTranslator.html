001    package com.gwt.ss;
002    
003    import javax.servlet.ServletContext;
004    
005    import org.aspectj.lang.ProceedingJoinPoint;
006    import org.aspectj.lang.annotation.Around;
007    import org.aspectj.lang.annotation.Aspect;
008    import org.slf4j.Logger;
009    import org.slf4j.LoggerFactory;
010    import org.springframework.context.ApplicationListener;
011    import org.springframework.security.access.event.AuthorizationFailureEvent;
012    import org.springframework.web.context.ServletContextAware;
013    
014    /**
015     * Rquired
016     * <ul>
017     * <li>Intruduce aop naming space</li>
018     * <li>Declare &lt;aop:aspectj-autoproxy&gt;</li>
019     * <li>Instantiate this bean</li>
020     * </ul>
021     * 必須
022     * <ul>
023     * <li>導入 aop 名稱空間</li>
024     * <li>設定 &lt;aop:aspectj-autoproxy&gt;</li>
025     * <li>建立本類別實例</li>
026     * </ul>
027     */
028    @Aspect
029    public class GwtExceptionTranslator implements ServletContextAware, ApplicationListener<AuthorizationFailureEvent> {
030    
031        private static ThreadLocal<HttpHolder> httpHolder = new InheritableThreadLocal<HttpHolder>();
032    
033        protected static final Logger LOG = LoggerFactory.getLogger(GwtExceptionTranslator.class);
034    
035        private ServletContext servletContext;
036    
037        @Around("execution(* org.springframework.security.web.access.ExceptionTranslationFilter.doFilter(..))")
038        public Object doFilter(ProceedingJoinPoint pjp) throws Throwable {
039            HttpHolder holder = HttpHolder.getInstance(pjp);
040            if (LOG.isDebugEnabled()) {
041                LOG.debug("Capture of " + (holder.isGwt() ? "Gwt" : "Non Gwt") + " ExceptionTranslationFilter.doFilter!"
042                        + " with " + holder.getRequest().getRequestURI());
043            }
044            if (holder.isGwt()) {
045                httpHolder.set(holder);
046                try {
047                    return pjp.proceed();
048                } finally {
049                    httpHolder.remove();
050                }
051            } else {
052                return pjp.proceed();
053            }
054        }
055    
056        /** {@inheritDoc} */
057        @Override
058        public void onApplicationEvent(AuthorizationFailureEvent event) {
059            if (LOG.isErrorEnabled()) {
060                LOG.error("Receive AuthorizationFailureEvent:" + event.getAccessDeniedException().getMessage(),
061                    event.getAccessDeniedException());
062            }
063            HttpHolder holder = httpHolder.get();
064            if (holder != null && holder.isGwt()) {
065                GwtResponseUtil.processGwtException(servletContext, holder.getRequest(), holder.getResponse(),
066                    event.getAccessDeniedException());
067            }
068        }
069    
070        /** {@inheritDoc} */
071        @Override
072        public void setServletContext(ServletContext servletContext) {
073            this.servletContext = servletContext;
074        }
075    
076    }
