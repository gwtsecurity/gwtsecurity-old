024    /**
025     * Utility for handle GWT RPC response.<br/>
026     * GWT RPC 回應處理工具類別
027     */
028    public final class GwtResponseUtil {
029    
030        private static AuthenticationTrustResolver authenticationTrustResolver = new AuthenticationTrustResolverImpl();
031    
032        private static final Logger LOG = LoggerFactory.getLogger(GwtResponseUtil.class);
033    
034        private static Class<?>[] securityClasses = null;
035    
036        /**
037         * @param ex the gwt exception.
038         * @return the gwt equivalent exception.
039         */
040        public static GwtSecurityException createGwtException(Exception ex) {
041            if (ex == null) { return null; }
042            String className = ex.getClass().getSimpleName();
043            try {
044                for (Class<?> rawClass : getSecurityClasses()) {
045                    Class<? extends GwtSecurityException> securityClass = rawClass.asSubclass(GwtSecurityException.class);
046                    String gwtClassName = securityClass.getSimpleName();
047                    if (gwtClassName.equals("Gwt" + className)) {
048                        Constructor<? extends GwtSecurityException> constructor = securityClass.getConstructor(
049                            String.class, Throwable.class);
050                        return constructor.newInstance(ex.getMessage(), ex);
051                    }
052                }
053            } catch (Exception e) {
054                LOG.error(e.getMessage(), e);
055            }
056            // Custom Mappings.
057            // **** Provide any custom mappings here ****
058    
059            // 沒有可用的映射 No mapping available
060            return null;
061        }
062    
063        /**
064         * @param response the response.
065         * @param e the error.
066         */
067        public static void doUnexpectedFailure(HttpServletResponse response, Throwable e) {
068            if (LOG.isErrorEnabled()) {
069                LOG.error("Encode eunexceptable exception!", e);
070            }
071            if (!response.isCommitted()) {
072                try {
073                    response.setContentType("text/plain");
074                    response.setStatus(HttpServletResponse.SC_INTERNAL_SERVER_ERROR);
075                    response.getWriter().write(e.getMessage());
076                    response.flushBuffer();
077                } catch (IOException ex) {
078                    if (LOG.isErrorEnabled()) {
079                        LOG.error(ex.getMessage(), ex);
080                    }
081                }
082            }
083        }
084    
085        /**
086         * @return the available security exception classes in the com.gwt.ss.client.exceptions package.
087         */
088        private static synchronized Class<?>[] getSecurityClasses() {
089            if (securityClasses == null) {
090                try {
091                    securityClasses = ClassUtil.getClasses("com.gwt.ss.client.exceptions");
092                } catch (Exception e) {
093                    if (LOG.isErrorEnabled()) {
094                        LOG.error(e.getMessage(), e);
095                    }
096                }
097            }
098            return securityClasses;
099        }
100    
101        /**
102         * @param authentication the current authentication.
103         * @return is the user anonymous?
104         */
105        public static boolean isAnonymous(Authentication authentication) {
106            return authenticationTrustResolver.isAnonymous(authentication);
107        }
108    
109        /**
110         * Determine whether request comes from GWT.
111         * 
112         * @param request the request.
113         * @return is the request from GWT RPC?
114         */
115        public static boolean isGwt(HttpServletRequest request) {
116            if (request == null) { return false; }
117            // Check for the request factory header.
118            String header = request.getHeader(GwtConst.GWT_RF_HEADER);
119            if (header != null && header.equalsIgnoreCase("true")) { return true; }
120            // Check for the gwt-rpc content type.
121            String contentType = request.getContentType();
122            if (contentType != null && contentType.startsWith(GwtConst.GWT_RPC_CONTENT_TYPE)) { return true; }
123            return false;
124        }
125    
126        public static void processGwtException(ServletContext servletContext, HttpServletRequest request,
127                HttpServletResponse response, Exception ex) {
128            try {
129                GwtSecurityException gwtEx = createGwtException(ex);
130                if (gwtEx != null) {
131                    // Get the authentication information.
132                    Authentication authentication = SecurityContextHolder.getContext().getAuthentication();
133                    boolean anonymous = authentication == null || isAnonymous(authentication);
134                    if (!anonymous) {
135                        // Add authentications information to the exception.
136                        gwtEx.setAuthenticated(authentication.isAuthenticated());
137                        gwtEx.setAuthenticatedUser(authentication.getName());
138                    }
139                    if (LOG.isErrorEnabled()) {
140                        String extra = "";
141                        if (ex instanceof AccessDeniedException) {
142                            extra = String.format(" (user is%1$s anonymous)", anonymous ? "" : " not");
143                        }
144                        LOG.error("Encode {}{}: {}",
145                            new Object[] { gwtEx.getClass().getSimpleName(), extra, gwtEx.getMessage() });
146                    }
147                }
148                String payload = RPC.encodeResponseForFailure(null, gwtEx == null ? ex : gwtEx);
149                writeResponse(servletContext, request, response, payload);
150            } catch (SerializationException ex1) {
151                doUnexpectedFailure(response, ex);
152            }
153        }
154    
155        public static boolean shouldCompressResponse(String responsePayload) {
156            return RPCServletUtils.exceedsUncompressedContentLengthLimit(responsePayload);
157        }
158    
159        public static void writeResponse(ServletContext servletContext, HttpServletRequest request,
160                HttpServletResponse response, String responsePayload) {
161            if (!response.isCommitted()) {
162                try {
163                    boolean gzipEncode = RPCServletUtils.acceptsGzipEncoding(request)
164                            && shouldCompressResponse(responsePayload);
165                    RPCServletUtils.writeResponse(servletContext, response, responsePayload, gzipEncode);
166                    response.flushBuffer();
167                } catch (IOException ex) {
168                    doUnexpectedFailure(response, ex);
169                }
170            }
171        }
172    
173        /**
174         * Private constructor.
175         */
176        private GwtResponseUtil() {
177            super();
178        }
