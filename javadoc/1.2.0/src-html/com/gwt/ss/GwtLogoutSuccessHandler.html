001    package com.gwt.ss;
002    
003    import java.io.IOException;
004    
005    import javax.servlet.ServletContext;
006    import javax.servlet.ServletException;
007    import javax.servlet.http.HttpServletRequest;
008    import javax.servlet.http.HttpServletResponse;
009    
010    import org.springframework.beans.factory.InitializingBean;
011    import org.springframework.security.core.Authentication;
012    import org.springframework.security.web.authentication.logout.LogoutSuccessHandler;
013    import org.springframework.security.web.authentication.logout.SimpleUrlLogoutSuccessHandler;
014    import org.springframework.util.Assert;
015    import org.springframework.web.context.ServletContextAware;
016    
017    import com.google.gwt.user.client.rpc.impl.AbstractSerializationStream;
018    
019    /**
020     * Assign to &lt;logout success-handler-ref=&quot;hander-reference&quot;/&gt; fot handle logout action.<br/>
021     * When logout from GWT RPC, just void message return. If logout from normal URL will redirect page to
022     * {@link #setLogoutSuccessUrl(java.lang.String) logoutSuccessUrl}.<br/>
023     * 建立本類別實例並設定到&lt;logout success-handler-ref=&quot;實例參照&quot;/&gt; 以處理登錄作業，<br/>
024     * 當使用GWT RPC進行登出時，本顯別會回應GWT RPC程式。<br/>
025     * 若使用正常址址登出，則頁面會回轉到{@link #setLogoutSuccessUrl(java.lang.String) logoutSuccessUrl}
026     */
027    public class GwtLogoutSuccessHandler implements LogoutSuccessHandler, InitializingBean, ServletContextAware {
028    
029        private SimpleUrlLogoutSuccessHandler defaultHandler = new SimpleUrlLogoutSuccessHandler();
030    
031        private String logoutSuccessUrl = "/";
032    
033        private ServletContext servletContext;
034    
035        @Override
036        public void afterPropertiesSet() {
037            Assert.notNull(logoutSuccessUrl, "logoutSuccessUrl must be specified");
038            defaultHandler.setDefaultTargetUrl(logoutSuccessUrl);
039        }
040    
041        public String getLogoutSuccessUrl() {
042            return logoutSuccessUrl;
043        }
044    
045        @Override
046        public void onLogoutSuccess(HttpServletRequest request, HttpServletResponse response, Authentication authentication)
047                throws IOException, ServletException {
048            if (GwtResponseUtil.isGwt(request)) {
049                String responsePayload = String.format("//OK[[],%d,%s]", AbstractSerializationStream.DEFAULT_FLAGS,
050                    AbstractSerializationStream.SERIALIZATION_STREAM_VERSION);
051                GwtResponseUtil.writeResponse(servletContext, request, response, responsePayload);
052            } else {
053                defaultHandler.onLogoutSuccess(request, response, authentication);
054            }
055    
056        }
057    
058        public void setLogoutSuccessUrl(String logoutSuccessUrl) {
059            this.logoutSuccessUrl = logoutSuccessUrl;
060            defaultHandler.setDefaultTargetUrl(logoutSuccessUrl);
061        }
062    
063        @Override
064        public void setServletContext(ServletContext servletContext) {
065            this.servletContext = servletContext;
066        }
067    }
