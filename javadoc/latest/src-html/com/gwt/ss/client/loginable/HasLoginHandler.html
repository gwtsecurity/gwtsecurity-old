001    package com.gwt.ss.client.loginable;
002    
003    import com.google.gwt.event.shared.HandlerRegistration;
004    import com.google.gwt.event.shared.HasHandlers;
005    
006    /**
007     * A interface interact with front user for processing login action, 
008     * and add a {@link com.gwt.ss.client.loginable.LoginHandler LoginHandler} to receive 
009     * login result.<br/>
010     * <span style="color:red">Important:</span> 
011     * when {@link AbstractLoginBox#startLogin(java.lang.Throwable) start Login}}, no matter login succeeded or canceled,
012     * It must fire a {@link LoginEvent} to notify the waiting {@link LoginableService}. 
013     * <br/>實做此界面之物件可讓用戶啟始登錄作業，並透過
014     * {@link com.gwt.ss.client.loginable.LoginHandler LoginHandler}接收登錄結果 <br/>
015     * <span style="color:red">注意:</span> 一旦{@link AbstractLoginBox#startLogin(java.lang.Throwable) 啟動登錄}，無論
016     * 登錄成功或用戶取消登錄，都一定要發送{@link LoginEvent}給等待中的{@link LoginableService}.
017     * @author Kent Yeh
018     */
019    public interface HasLoginHandler extends HasHandlers {
020        /**
021         * start login action
022         * 開始啟始登錄作業
023         * @param caught <table style="padding-left:70px"><tr><td>then reason why should start login.</td></td>
024         *<tr><td>啟動登錄作業的異常例外/td></td></table>
025         */
026        public void startLogin(Throwable caught);
027        /**
028         * Accept {@link LoginHandler} to handle {@link LoginEvent}<br/>
029         * 接受 {@link LoginHandler} 以處理 {@link LoginEvent}
030         */
031        public HandlerRegistration addLoginHandler(LoginHandler handler);
032    }
