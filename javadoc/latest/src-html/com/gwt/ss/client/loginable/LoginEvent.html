001    package com.gwt.ss.client.loginable;
002    
003    import com.google.gwt.event.shared.GwtEvent;
004    
005    /**
006     * Represents login process result.<br/>
007     * 表示登錄的最後結果  
008     * @author Kent Yeh
009     */
010    public class LoginEvent extends GwtEvent<LoginHandler> {
011    
012        private boolean succeeded = false;
013    
014        /**
015         * 
016         * @param succeeded <table style="padding-left:70px"><tr><td>true meeas Login success, otherwise login is canceled.</td></td>
017         *<tr><td>true 表示登錄成功，否則為用戶取消登錄作業</td></td></table>
018         */
019        public LoginEvent(boolean succeeded) {
020            this.succeeded = succeeded;
021        }
022        private static Type<LoginHandler> type = new Type<LoginHandler>();
023    
024        /**
025         * Does user cancel loggin process?<br/>
026         * 是否用戶取消登錄?     
027         */
028        public boolean isCanceled() {
029            return !succeeded;
030        }
031    
032        /**
033         * Does user login success?<br/>
034         * 是否用戶已登錄成功?     
035         */
036        public boolean isLoginSuccessful() {
037            return succeeded;
038        }
039    
040        @Override
041        protected void dispatch(LoginHandler handler) {
042            handler.onLogin(this);
043        }
044    
045        @Override
046        public Type<LoginHandler> getAssociatedType() {
047            return type;
048        }
049    
050        public static Type<LoginHandler> getType() {
051            return type;
052        }
053    }
