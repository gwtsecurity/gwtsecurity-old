001    package com.gwt.ss.client;
002    
003    import com.google.gwt.core.client.GWT;
004    import com.google.gwt.user.client.rpc.AsyncCallback;
005    
006    public interface GwtLogoutAsync {
007    
008        /**
009         * Maintain singleton {@link GwtLogoutAsync GwtLogoutAsync} instance.<br/>
010         * 管控{@link GwtLogoutAsync GwtLogoutAsync}單一實例工具
011         */
012         static final class Util {
013    
014            private static final AsyncUtil<GwtLogoutAsync> util = new AsyncUtil<GwtLogoutAsync>("j_spring_security_logout") {
015                @Override
016                protected GwtLogoutAsync newInstance() {
017                    return (GwtLogoutAsync) GWT.create(GwtLogout.class);
018                }
019            };
020    
021            /**
022             * return {@link GwtLoginAsync GwtLoginAsync} instance with default service entry point
023             * &quot;j_spring_security_check&quot;<br/>
024             * 取得預設進入點為&quot;j_spring_security_check&quot;之{@link GwtLoginAsync GwtLoginAsync} 實例
025             * @return the instance.
026             */
027            public static GwtLogoutAsync getInstance() {
028                return util.getInstance();
029            }
030    
031            /**
032             * return {@link GwtLoginAsync GwtLoginAsync} instance with specified service entry point.<br/>
033             * 指定服務進入點並取得{@link GwtLoginAsync GwtLoginAsync} 實例
034             * @param processUrl Spring login processing url
035             * @return the instance.
036             */
037            public static GwtLogoutAsync getInstance(String processUrl) {
038                return util.getInstance(processUrl);
039            }
040    
041            public static String getProcessUrl() {
042                return util.getProcessUrl();
043            }
044    
045            public static void setProcessUrl(String processUrl) {
046                util.setProcessUrl(processUrl);
047            }
048    
049            private Util() {
050                // Utility class should not be instantiated.
051            }
052    
053        }
054    
055        void j_gwt_security_logout(AsyncCallback<Void> callback);
056    
057    }
