001    package com.gwt.ss.client.loginable;
002    
003    /**
004     * A wrapper(proxy) for asynchronous {@link com.google.gwt.user.client.rpc.RemoteService remoteService}
005     * descended from {@link LoginableAsync LoginableAsync}.<br/>
006     * 繼承自{@link LoginableAsync LoginableAsync}的非同步{@link com.google.gwt.user.client.rpc.RemoteService remoteService}
007     * 的代理物件。 
008     * @author Kent Yeh
009     */
010    public interface LoginableService<T extends LoginableAsync> {
011    
012        HasLoginHandler getHasLoginHandler();
013    
014        void setHasLoginHandler(HasLoginHandler hasLoginHandler);
015    
016        /**
017         * return then real asynchronous {@link com.google.gwt.user.client.rpc.RemoteService remoteService}, If not 
018         * setup before, generate by {@link com.google.gwt.core.client.GWT#create(java.lang.Class create)}
019         *  automaticly.<br/>
020         * 回傳之前設定的非同步{@link com.google.gwt.user.client.rpc.RemoteService remoteService}，若是未曾設定，則叫用
021         * {@link com.google.gwt.core.client.GWT#create(java.lang.Class create)}
022         * 自動建立一個
023         */
024        T getRemoteService();
025    
026        /**
027         * setup then real asynchronous {@link com.google.gwt.user.client.rpc.RemoteService remoteService}<br/>
028         * 設定非同步{@link com.google.gwt.user.client.rpc.RemoteService remoteService}.
029         */
030        void setRemoteService(T service);
031    }
