003    /**
004     * Whenever user cancel login process occurence.<br/>
005     * 用戶取消登錄作業時觸發的錯誤            
006     * @author Kent Yeh
007     */
008    public class LoginCancelException extends Exception {
009    
010        private static final long serialVersionUID = -1265673524265096271L;
011    
012        public LoginCancelException(Throwable cause) {
013            super(cause);
014        }
015    
016        public LoginCancelException(String message, Throwable cause) {
017            super(message, cause);
018        }
019    
020        public LoginCancelException(String message) {
021            super(message);
022        }
023    
024        public LoginCancelException() {
025        }
026    }
